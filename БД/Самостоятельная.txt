SELECT Salespeople.snum, Salespeople.sname, COUNT(*) AS 'Кол-во заказов'
FROM Salespeople, Orders
WHERE Salespeople.snum = Orders.snum
GROUP BY Salespeople.snum, Salespeople.sname
HAVING COUNT(*)>2

2.
SELECT customers.cnum, customers.cname, orders.odate, COUNT(*) 'Количество заказов'
FROM customers, orders
WHERE customers.cnum = orders.cnum
GROUP BY customers.cnum, customers.cname
HAVING COUNT(*)>1

3.
SELECT city, SUM(rating)
FROM customers
GROUP BY city
HAVING SUM(rating) >= ALL(SELECT SUM(rating) FROM customers GROUP BY city)
